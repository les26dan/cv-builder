Phụ lục này hướng dẫn chi tiết cách cài đặt và chạy hệ thống trên môi trường phát triển cục bộ.

\section{Yêu cầu hệ thống}

\begin{itemize}
\item Node.js phiên bản 18.x trở lên
\item npm phiên bản 9.x trở lên
\item Tài khoản Supabase (miễn phí tại \url{https://supabase.com})
\item Tài khoản OpenAI với API key (tại \url{https://platform.openai.com})
\end{itemize}

\section{Cài đặt dự án}

\begin{lstlisting}[language=bash, caption={Các bước cài đặt dự án}]
# Clone repository
git clone <repository-url>
cd pmf

# Cai dat dependencies
npm install

# Tao file cau hinh moi truong
cp docs/env-template.txt .env.local
\end{lstlisting}

\section{Cấu hình biến môi trường}

Chỉnh sửa file \texttt{.env.local} với các thông tin sau:

\begin{lstlisting}[language=bash, caption={Các biến môi trường cần cấu hình}]
# Supabase
NEXT_PUBLIC_SUPABASE_URL=https://your-project.supabase.co
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key

# Session secret
SESSION_SECRET=your-session-secret

# OpenAI
NEXT_PUBLIC_OPENAI_API_KEY=your-openai-api-key

# Vercel Blob
BLOB_READ_WRITE_TOKEN=your-blob-token
\end{lstlisting}

\section{Thiết lập cơ sở dữ liệu}

\begin{enumerate}
\item Tạo dự án mới trên Supabase Dashboard
\item Mở SQL Editor trong Supabase Dashboard
\item Chạy nội dung file \texttt{docs/database-schema.sql} để tạo các bảng, index, RLS policies và functions
\item Xác nhận tất cả các bảng đã được tạo thành công: \texttt{users}, \texttt{cv\_workflow}, \texttt{cv\_drafts}, \texttt{job\_postings}, \texttt{job\_applications}, \texttt{cv\_job\_searches} (xem ERD ở Hình~\ref{fig:erd}).
\end{enumerate}

\section{Chạy ứng dụng}

\begin{lstlisting}[language=bash, caption={Khởi động ứng dụng}]
# Chay moi truong phat trien
npm run dev

# Truy cap ung dung tai
# http://localhost:3000

# Build cho production
npm run build
npm run start

# Chay kiem thu
npm run test
\end{lstlisting}
